State-driven Compose with hoisted state CLO 2

Kotlin
@Composable
fun Counter() {
    // Survives recomposition; lost on configuration change unless saved.
    var count by remember { mutableStateOf(0) }

    // The UI is a function of state. Changing state triggers recomposition.
    Column(
        modifier = Modifier.padding(16.dp),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text(text = "Count: $count", style = MaterialTheme.typography.headlineSmall)
        Button(onClick = { count++ }) {   // event flows up, state flows down
            Text("Increment")
        }
    }
}

Room: entity, DAO returning a Flow, suspend write CLO 4

Kotlin
@Entity(tableName = "notes")
data class Note(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    val title: String,
    val done: Boolean = false
)

@Dao
interface NoteDao {
    // Observable read: emits a new list whenever the table changes.
    @Query("SELECT * FROM notes ORDER BY id DESC")
    fun observeAll(): Flow<List<Note>>

    // Writes must be suspend so Room runs them off the main thread.
    @Insert suspend fun insert(note: Note)
    @Delete suspend fun delete(note: Note)
}

Retrofit + coroutines: networking off the main thread CLO 5CLO 6

Kotlin
// 1) Model that matches the JSON the API returns.
data class Weather(val city: String, val tempC: Double)

// 2) Retrofit service: suspend fun = runs on a background dispatcher.
interface WeatherApi {
    @GET("weather")
    suspend fun getWeather(@Query("city") city: String): Weather
}

// 3) Repository is the single boundary the ViewModel talks to.
class WeatherRepository(private val api: WeatherApi) {
    suspend fun load(city: String): Weather = api.getWeather(city)
}

// 4) ViewModel launches in viewModelScope and exposes state.
class WeatherViewModel(private val repo: WeatherRepository) : ViewModel() {
    var uiState by mutableStateOf<String>("Loading…")
        private set

    fun refresh(city: String) {
        viewModelScope.launch {
            uiState = try {
                val w = repo.load(city)   // suspends, does not block the UI thread
                "${'$'}{w.city}: ${'$'}{w.tempC}°C"
            } catch (e: Exception) {
                "Error: ${'$'}{e.message}"
            }
        }
    }
}

Navigation between two screens (Navigation-Compose) CLO 2CLO 3

Kotlin
@Composable
fun AppNav() {
    val navController = rememberNavController()
    NavHost(navController = navController, startDestination = "list") {
        composable("list") {
            ListScreen(onItemClick = { id -> navController.navigate("detail/$id") })
        }
        composable("detail/{id}") { backStackEntry ->
            val id = backStackEntry.arguments?.getString("id")
            DetailScreen(id = id)
        }
    }
}
💡
One-line architecture map UI (Compose) → ViewModel (state + viewModelScope) → Repository → (Room / DataStore for local, Retrofit for network). Several questions test whether you know which layer owns which job.
